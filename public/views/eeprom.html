      #define default_BUMP_DEBOUNCE_TIME_ms      20

      /* be calculated as v = (1g -FREEFALL_THRESHOLD)*FREEFALL_DEBOUNCE_TIME  */

      /* Test 1: v = 1.37 m/s

      */

      #define default_FREEFALL_DEBOUNCE_TIME_ms         250
